Convert to a PNG QR image base64 for frontend. // Simpler: send the raw QR str and frontend can render it as text or use a QR library. broadcast({ type:'qr', data: 'data:image/png;base64,' + Buffer.from(qr).toString('base64') }); broadcast({ type:'status', data:'qr' }); } if(connection === 'open'){ broadcast({ type:'status', data:'open' }); // Send session file content to frontend try{ if(fs.existsSync(authFile)){ const raw = fs.readFileSync(authFile,'utf8'); let json; try{ json = JSON.parse(raw); }catch(e){ json = { raw }; } broadcast({ type:'session', data: json }); } }catch(e){ console.error(e); } } if(lastDisconnect && lastDisconnect.error){ const code = lastDisconnect.error?.output?.statusCode || lastDisconnect.error?.statusCode || 0; console.log('disconnected', code); broadcast({ type:'status', data:'disconnected' }); // try reconnect if(sock){ try{ sock.end(); }catch(e){} sock = makeWASocket({ auth: state, version }); } } }); // WS messages wss.on('connection', ws=>{ ws.on('message', msg=>{ try{ const o = JSON.parse(msg.toString()); if(o.action === 'refreshQR'){ // a safe way to force a new QR is to logout and re-initialize auth console.log('refresh requested'); try{ sock.logout(); }catch(e){} } }catch(e){ } }); }); server.listen(PORT, ()=>console.log('Listening on', PORT)); })(); --- backend/Dockerfile (optional) FROM node:18-alpine WORKDIR /app COPY package*.json ./ RUN npm ci --only=production COPY . .
